Data Analytics and Visualisation.

The purpose of this course is to develop advanced knowledge and skills in data analytics and data wrangling for effective data-driven decision making and data visualisation.

Tools: RStudio

sample 1: Pie Chart of App Category in App Store


appnum <- appstore %>%
  group_by(prime_genre) %>%
  summarise(appnum = n()) 

p1 <- nPlot(appnum ~ prime_genre, data = appnum, type = 'pieChart')
p1$templates$script <- "http://timelyportfolio.github.io/rCharts_nvd3_templates/chartWithTitle.html"
p1$set(title = "Pie Chart of App Category in App Store")
p1$xAxis(axisLabel = 'Apps Category')
p1$chart(donut = FALSE)
p1

sample 2: Free and Paid Apps by Category

//get the total number of every category
appnum <- appstore %>%
  group_by(prime_genre) %>%
  summarise(appnum = n()) 

paid <- filter(appstore,price > 0)
free <- filter(appstore,price == 0)

paidApp <- paid %>%
  group_by(prime_genre) %>%
  summarise(num = n()) %>%
  mutate(type = "Paid Apps")

freeApp <- free %>%
  group_by(prime_genre) %>%
  summarise(num = n()) %>%
  mutate(type = "Free Apps")

total <- rbind(paidApp,freeApp)

p2 = nPlot(num ~ prime_genre, group = 'type', data = total, type = 'multiBarChart')
p2$chart(color = c('blue', 'red'))
p2$templates$script <- "http://timelyportfolio.github.io/rCharts_nvd3_templates/chartWithTitle.html"
p2$set(title = "Free Apps and Paid Apps in App Store")
p2$xAxis(axisLabel = 'Apps Category')
p2
p2$save('p2.html',cdn=TRUE)

sample 3: Pie Chart

appnum <- appstore %>%
  group_by(prime_genre) %>%
  summarise(appnum = n()) 

paid <- filter(appstore,price > 0)
free <- filter(appstore,price == 0)

paidApp <- paid %>%
  group_by(prime_genre) %>%
  summarise(num = n()) %>%
  mutate(type = "Paid Apps")

freeApp <- free %>%
  group_by(prime_genre) %>%
  summarise(num = n()) %>%
  mutate(type = "Free Apps")

#hPlot画全局饼图
#x，y对应横纵坐标
#data 数据源
#type 显示类型
total <- rbind(paidApp,freeApp)
p1 <- hPlot(x = "prime_genre", y = "appnum", data = appnum, type = "pie")
p1
#画分类饼图
cat <- c("Book", "Business", "Catalogs", "Education", "Entertainment", "Finance", "Food & Drink", "Games", "Health & Fitness", "Lifestyle", "Medical", "Music", "Navigation", "News", "Photo & Video", "Productivity", "Reference", "Shopping", "Social Networking", "Sports", "Travel", "Utilities", "Weather") 
pielist <- vector("list",23) 
index <- c(1:23)
for(i in index){
  item <- filter(total, prime_genre == cat[i])
  pielist[[i]] <- item
}
#饼图风格1 hPlot
p2 <- hPlot(x = "type", y = "num", data = pielist[[8]], type = "pie")
p2$chart(donut = TRUE)
p2
#饼图风格2 nPlot
p2 <- nPlot(num ~ type, data = pielist[[1]], type = 'pieChart')
p2$chart(donut = TRUE)
p2

sample 4: Scatter Chart

apps <- filter(appstore, appstore$price<=100)
apps <- apps %>% mutate(type = ifelse(apps$price>0,"Paid Apps","Free Apps"))
apps$size_mb <- apps$size_bytes/1024/1024
apps <- head(apps,4000)
p4 <- nPlot(size_mb ~price, group = 'type', data = apps, type = 'scatterChart')
p4$xAxis(axisLabel = 'Price')
p4
